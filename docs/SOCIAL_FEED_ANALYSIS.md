**3. Person/Institution Filtering**

```typescript
// State
const [filter, setFilter] = useState<Filter>({
  type: null, // 'person' | 'institution' | null
  value: null
});

// Filter logic
const filteredTransactions = useMemo(() => {
  if (!filter.type) return transactions;

  return transactions.filter(t => {
    if (filter.type === 'person') return t.personId === filter.value;
    if (filter.type === 'institution') return t.institution === filter.value;
  });
}, [transactions, filter]);

// UI
<Chip
  avatar={<Avatar.Image source={person.avatar} />}
  onPress={() => setFilter({ type: 'person', value: person.id })}
>
  {person.name}
</Chip>
```

**Complexity: VERY LOW** (basic filtering)
